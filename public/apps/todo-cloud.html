// 雲端待辦事項 - Cloudflare KV 儲存 let todos = []; let userId = localStorage.getItem('jhm5_4_cloud_user') || ''; async function loadTodos() { if (!userId) return; try { const response = await fetch(`/api/todos/${userId}`); todos = await response.json(); render(); } catch (error) { console.error('載入失敗:', error); alert('載入待辦事項失敗，請檢查網路連線'); } } async function saveTodos() { if (!userId) return; try { await fetch(`/api/todos/${userId}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(todos) }); } catch (error) { console.error('儲存失敗:', error); } } function setUser() { const input = document.getElementById('userInput'); const newUserId = input.value.trim(); if (!newUserId) { alert('請輸入使用者 ID'); return; } userId = newUserId; localStorage.setItem('jhm5_4_cloud_user', userId); loadTodos(); } async function addTodo() { const text = document.getElementById('todoInput').value.trim(); if (!text) return; todos.push({ id: Date.now(), text, completed: false }); await saveTodos(); render(); }
